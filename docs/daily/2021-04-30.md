# Leet code - 1214 two-pointer-technique

    ```c++
    /**
     * Definition for singly-linked list.
     * struct ListNode {
     *     int val;
     *     ListNode *next;
     *     ListNode(int x) : val(x), next(NULL) {}
     * };
     */
    class Solution {
    public:
        ListNode *detectCycle(ListNode *head) {
            if(head==nullptr)
                return nullptr;
            if(head->val==-1)
                return nullptr;
            if(head->next == nullptr || head->next->next == nullptr)
                return nullptr;
            // find isCycle
            int isCycle=false;
            ListNode* ptrFast=head->next->next;
            ListNode* ptrSlow=head->next;
            while(1)
            {
                if(ptrFast->next==nullptr||ptrFast->next->next==nullptr)
                    return nullptr;
                if(ptrSlow==ptrFast)
                    break;
                ptrSlow=ptrSlow->next;
                ptrFast=ptrFast->next->next;
            }
            ptrSlow = head->next;
            ptrFast = ptrFast->next;
            while(1)
            {
                if(ptrFast==ptrSlow)
                    return ptrSlow;
                ptrSlow=ptrSlow->next;
                ptrFast=ptrFast->next;
            }
            
            return nullptr;
        }
    };
    ```
