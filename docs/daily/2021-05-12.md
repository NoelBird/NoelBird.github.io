## 3157 - Move Zeroes

3분 정도 소요



=== "C++ time(beats 89.67%), space(beats 55.77%)"

    ```c++
    class Solution {
    public:
        void moveZeroes(vector<int>& nums) {
            int size = nums.size();
            int idxNotZero = 0;
            int tmp=0;
            for(int i=0; i<size; ++i)
            {
                if(nums[i]!=0)
                {
                    nums[idxNotZero] = nums[i];
                    idxNotZero++;
                }
            }
            for(int i=idxNotZero; i<size; ++i)
            {
                nums[i] = 0;
            }
        }
    };
    ```
